Project arbit {
  database_type: 'MySQL'
  Note: 'Arbit database structure based on the current JPA entities.'
}

Enum user_gender {
  MALE
  FEMALE
  NONSELECT
}

Enum event_status {
  UPCOMING
  ONGOING
  CLOSED
}

Enum keyword_type {
  CATEGORY
  MOOD
  TOPIC
  AUDIENCE
  AGE
  FEATURE
}

Enum event_action_type {
  DETAIL_VIEW
  HOMEPAGE_CLICK
}

Enum event_action_source {
  HOME
  SEARCH
  RECOMMENDATION
  EVENT_DETAIL
}

Table users {
  id binary(16) [pk]
  username varchar(50) [not null, unique]
  password varchar(255) [not null]
  nickname varchar(100)
  profile_image_url varchar(1000)
  age int
  gender user_gender
  residential_area varchar(255)
  residential_latitude double
  residential_longitude double
  created_at datetime [not null]
  updated_at datetime
}

Table categories {
  id bigint [pk, increment]
  name varchar(50) [not null, unique]
  created_at datetime [not null]
  updated_at datetime
}

Table user_categories {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  category_id bigint [not null, ref: > categories.id]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, category_id) [unique, name: 'uk_user_category']
  }
}

Table events {
  id binary(16) [pk]
  category_id bigint [not null, ref: > categories.id]
  title varchar(200) [not null]
  description text
  poster_image_url varchar(1000)
  venue varchar(100) [not null]
  venue_address varchar(255)
  district varchar(50) [not null]
  latitude double
  longitude double
  start_date date [not null]
  end_date date [not null]
  free boolean [not null]
  status event_status [not null]
  average_rating decimal(3,2) [not null]
  price varchar(255)
  booking_url varchar(1000)
  created_at datetime [not null]
  updated_at datetime
}

Table bookmarks {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, ref: > events.id]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, event_id) [unique, name: 'uk_bookmark_user_event']
  }
}

Table reviews {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, ref: > events.id]
  rating int [not null, note: 'Application validation range: 1 to 5']
  content varchar(200) [not null]
  verification_image_url varchar(500)
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, event_id) [unique, name: 'uk_review_user_event']
  }
}

Table preference_keywords {
  id bigint [pk, increment]
  value varchar(50) [not null, unique]
  created_at datetime [not null]
  updated_at datetime
}

Table event_keywords {
  id bigint [pk, increment]
  event_id binary(16) [not null, ref: > events.id]
  preference_keyword_id bigint [not null, ref: > preference_keywords.id]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (event_id, preference_keyword_id) [unique, name: 'uk_event_keyword']
  }
}

Table classification_keywords {
  id bigint [pk, increment]
  value varchar(50) [not null, unique]
  created_at datetime [not null]
  updated_at datetime
}

Table event_classifications {
  id bigint [pk, increment]
  event_id binary(16) [not null, ref: > events.id]
  classification_keyword_id bigint [not null, ref: > classification_keywords.id]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (event_id, classification_keyword_id) [unique, name: 'uk_event_classification']
  }
}

Table age_restriction_keywords {
  id bigint [pk, increment]
  value varchar(50) [not null, unique]
  created_at datetime [not null]
  updated_at datetime
}

Table event_age_restrictions {
  id bigint [pk, increment]
  event_id binary(16) [not null, unique, ref: > events.id]
  age_restriction_keyword_id bigint [not null, ref: > age_restriction_keywords.id]
  created_at datetime [not null]
  updated_at datetime
}

Table recommendations {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, ref: > events.id]
  match_score decimal(5,2) [not null]
  reason varchar(500) [not null]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, event_id) [unique, name: 'uk_recommendation_user_event']
  }
}

Table user_preference_events {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, note: 'Stores an event UUID, but the JPA entity does not define a foreign-key relationship.']
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, event_id) [unique, name: 'uk_user_preference_event']
  }
}

Table keywords {
  id bigint [pk, increment]
  type keyword_type [not null]
  value varchar(80) [not null]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (type, value) [unique, name: 'uk_keyword_type_value']
  }
}

Table user_keyword_weights {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  keyword_id bigint [not null, ref: > keywords.id]
  weight decimal(5,4) [not null]
  source varchar(30) [not null]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, keyword_id) [unique, name: 'uk_user_keyword_weight']
  }
}

Table event_keyword_weights {
  id bigint [pk, increment]
  event_id binary(16) [not null, ref: > events.id]
  keyword_id bigint [not null, ref: > keywords.id]
  weight decimal(5,4) [not null]
  source varchar(30) [not null]
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (event_id, keyword_id) [unique, name: 'uk_event_keyword_weight']
  }
}

Table event_detail_view_logs {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, ref: > events.id]
  created_at datetime [not null]
  updated_at datetime
}

Table event_action_logs {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  event_id binary(16) [not null, ref: > events.id]
  action_type event_action_type [not null]
  source event_action_source
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (user_id, created_at) [name: 'idx_event_action_user_created']
    (event_id, action_type, created_at) [name: 'idx_event_action_event_type_created']
    (user_id, event_id, action_type) [name: 'idx_event_action_user_event_type']
  }
}

Table recommendation_runs {
  id bigint [pk, increment]
  user_id binary(16) [not null, ref: > users.id]
  algorithm varchar(50) [not null]
  model_version varchar(50) [not null]
  input_event_ids_json text
  created_at datetime [not null]
  updated_at datetime
}

Table recommendation_items {
  id bigint [pk, increment]
  run_id bigint [not null, ref: > recommendation_runs.id]
  event_id binary(16) [not null, ref: > events.id]
  rank_no int [not null]
  score decimal(8,4) [not null]
  reason varchar(300) [not null]
  feature_scores_json text
  created_at datetime [not null]
  updated_at datetime

  indexes {
    (run_id, event_id) [unique, name: 'uk_recommendation_item_run_event']
  }
}
